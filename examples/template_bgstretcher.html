plugin.tx_karussell_pi1.templateFile = EXT:karussell/examples/template_bgstretcher.html
plugin.tx_karussell_pi1.styleFile = fileadmin/template/styles/bgstretcher.css
plugin.tx_karussell_pi1.jsFile = fileadmin/template/files/bgstretcher.js
